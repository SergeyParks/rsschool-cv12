let array = [9,3,6,1,6,8,3,5,7,9,6,4,5,7]
function bubble(arr) {
    for(let x = 0; x < arr.length; x++) {
        for(let y = 0; y < arr.length; y++) {
            if (arr[y] > arr[y + 1] && (y + 1 < arr.length)) {
                let z = arr[y];
                arr[y] = arr[y +1];
                arr[y + 1] = z;
            }
        }
    }
    return arr
}
